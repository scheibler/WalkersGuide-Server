API
===

Walkersguide-Manager
--------------------

get_available_maps():
* input: None
* output: json
    * available_maps: Dict<Name:String, Port:Integer>


WalkersGuide-MapServer
----------------------

get_route():
* input: json via post
    * session_id: String
    * source_points: List<Point>
    + allowed_way_classes: List<String>
    + blocked_ways: List<Integer>
    + indirection_factor: Integer
    + language: String      ["de", "en"]
* output: gzipped json
    * description: String
    * route: List<Point,Segment,Point,...>
    + error: String

get_next_intersections_for_way():
* input: json via post
    * session_id: String
    * node_id: Long
    * way_id: Long
    * next_node_id: Long
    + language: String      ["de", "en"]
* output: gzipped json
    * next_intersections: List<Point>
    + error: String

get_poi():
* input: json via post
    * session_id: String
    * lat: Double
    * lon: Double
    * radius: Integer
    * number_of_results: Integer
    * tags: List<String>
    + language: String      ["de", "en"]
    + search: String
* output: gzipped json
    * poi: List<Point>
    + error: String

get_departures():
* input: json via post
    * lat: double
    * lon: Double
    + language: String      ["de", "en"]
    + vehicle_list: List<String>
    + public_transport_provider : String      ["de", "en"]
* output: gzipped json
    * departures: List<Departure>
    + error: String

cancel_request():
* input: json via post
    * session_id: String
    + language: String      ["de", "en"]
* output: gzipped json
    + error: String

get_status():
* input: None
* output: gzipped json
    * map_name: String
    * map_version: Integer
    * map_created: Long
    * supported_poi_tags: List<String>
    * supported_languages: Dict<Identifier:String,Name:String>
    * supported_public_transport_provider: Dict<Identifier:String,Name:String>
    * supported_indirection_factors: List<Integer>
    * supported_way_classes: List<String>
