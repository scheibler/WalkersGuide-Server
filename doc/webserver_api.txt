API
===



WalkersGuide-MapServer
----------------------

Serverversion: 1.2.0

get_route():
* input: json via post
    * map_id: String
    * session_id: String
    * source_points: List<Point>
    + allowed_way_classes: Dict<String:Double>
    + blocked_ways: List<Integer>
    + language: String      ["de", "en"]
* output: gzipped json
    * description: String
    * route: List<Point,Segment,Point,...>

get_next_intersections_for_way():
* input: json via post
    * map_id: String
    * session_id: String
    * node_id: Long
    * way_id: Long
    * next_node_id: Long
    + language: String      ["de", "en"]
* output: gzipped json
    * next_intersections: List<Point>

get_hiking_trails():
* input: json via post
    * map_id: String
    * session_id: String
    * lat: Double
    * lon: Double
    * radius: Integer
    + language: String      ["de", "en"]
* output: gzipped json
    * hiking_trails: List<HikingTrail>

get_poi():
* input: json via post
    * map_id: String
    * session_id: String
    * lat: Double
    * lon: Double
    * radius: Integer
    * number_of_results: Integer
    * tags: List<String>
    + language: String      ["de", "en"]
    + search: String
* output: gzipped json
    * poi: List<Point>

send_feedback():
* input: json via post
    * token:    String      See webserver/constants.py:supported_feedback_token_list
    * message:  String
    + sender:   String
* output: None

cancel_request():
* input: json via post
    * session_id: String
* output: None

get_status():
* input: None
* output: gzipped json
    * server_name: String
    * server_version: String
    * supported_api_version_list: List<Integer>
    * supported_map_version_list: List<Integer>
    * supported_language_list: List<String>             ["de", "en"]
    * supported_poi_category_list: List<String>
    * maps: Dict<map_id:String,map_data:Dict>
